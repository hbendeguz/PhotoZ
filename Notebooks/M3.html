def m_model():
    model = Sequential()

    model.add(Conv2D(48, (3, 3), input_shape=(50,50,5), name='L01', activation='relu'))
    
    model.add(Conv2D(48, (3, 3), name='L02', activation='relu'))

    model.add(AveragePooling2D(pool_size=(2, 2), name='L03'))
    model.add(Conv2D(96, (3, 3), name='L04', activation='relu'))
   
    model.add(Conv2D(96, (3, 3), name='L05', activation='relu'))
    
    model.add(Conv2D(96, (3, 3), name='L051', activation='relu'))
    
    model.add(AveragePooling2D(pool_size=(2, 2), name='L06'))
    
    model.add(Conv2D(200, (3, 3), name='L07', padding='same', activation='relu'))
    
    model.add(Conv2D(200, (3, 3), name='L08', padding='same',activation='relu'))
    model.add(AveragePooling2D(pool_size=(2, 2), name='L09'))

    model.add(Conv2D(200, (3, 3), name='L10', padding='same',activation='relu'))
    model.add(BatchNormalization())
    model.add(Conv2D(200, (3, 3), name='L11', padding='same',activation='relu'))
    model.add(AveragePooling2D(pool_size=(2, 2), name='L12'))
    
    model.add(Conv2D(200, (3, 3), name='L13', padding='same',activation='relu'))
    
    model.add(Conv2D(20, (3, 3), name='L14', padding='same',activation='relu'))
    model.add(Conv2D(72, (1, 1), name='L15', activation='relu'))
    
    model.add(Flatten())
    model.add(Dense(4096, name='L16', activation='relu'))
    model.add(Dense(400, name='L17', activation='relu'))

    model.add(Dense(1, name='L18'))
    model.compile(optimizer=optimizers.Adam(lr=1e-5, beta_1=0.9, beta_2=0.999), loss='mse')


    return model

def get_new_model():
    
    model = Sequential()

    model.add(Conv2D(32, (3, 3), input_shape=(50,50,5), name='L01', activation='relu'))
    model.add(Conv2D(32, (3, 3), name='L02', activation='relu'))

    model.add(AveragePooling2D(pool_size=(2, 2), name='L03'))
    model.add(Conv2D(64, (3, 3), name='L04', activation='relu'))
    model.add(Conv2D(64, (3, 3), name='L05', activation='relu'))
    
    model.add(AveragePooling2D(pool_size=(2, 2), name='L06'))
    
    model.add(Conv2D(128, (3, 3), name='L07', padding='same', activation='relu'))
    model.add(Conv2D(128, (3, 3), name='L08', padding='same',activation='relu'))
    model.add(AveragePooling2D(pool_size=(2, 2), name='L09'))

    model.add(Conv2D(128, (3, 3), name='L10', padding='same',activation='relu'))
    model.add(Conv2D(128, (3, 3), name='L11', padding='same',activation='relu'))
    model.add(AveragePooling2D(pool_size=(2, 2), name='L12'))
    
    model.add(Conv2D(128, (3, 3), name='L13', padding='same',activation='relu'))
    model.add(Conv2D(128, (3, 3), name='L14', padding='same',activation='relu'))
    model.add(Conv2D(32, (1, 1), name='L15', activation='relu'))
    
    model.add(Flatten())
    model.add(Dense(2048, name='L16', activation='relu'))
    model.add(Dense(256, name='L17', activation='relu'))

    model.add(Dense(1, name='L18'))
    model.compile(optimizer=optimizers.Adam(lr=1e-4, beta_1=0.9, beta_2=0.999), loss='mse')

    
    return model
