import keras
from keras.layers import Input, MaxPooling2D, Conv2D, Dense, Reshape, Flatten, Concatenate, AveragePooling2D, Dropout
from keras.models import Model, load_model
from keras import backend as K
#K.set_image_dim_ordering('tf')

# Headline input: meant to receive sequences of 100 integers, between 1 and 10000.
# Note that we can name any layer by passing it a "name" argument.
main_input = Input(shape = (50, 50, 5))

x = Conv2D(28 ,(3, 3), activation='relu', padding='same', name='block1_conv1')(main_input)
#x = Dropout(0.6)(x)
x = Conv2D(28, (3, 3), activation='relu', padding='same', name='block11_conv2')(x)
x = MaxPooling2D((2, 2), strides=(1,1), name='block0_pool')(x)
x = Conv2D(64, (3, 3), activation='relu', padding='same', name='block1_conv2')(x)
x = Conv2D(64, (3, 3), activation='relu', padding='same', name='block1_conv3')(x)


x = MaxPooling2D((2, 2), strides=(2, 2), name='block1_pool')(x)


#x = MaxPooling2D((2, 2), strides=(2, 2), name='block2_pool')(x)

# Block 3
x = Conv2D(128, (3, 3), activation='relu', padding='same', name='block3_conv0')(x)
x = Conv2D(128, (3, 3), activation='relu', padding='same', name='block3_conv1')(x)
x = MaxPooling2D((3,3))(x)
x = Conv2D(128, (3, 3), activation='relu', padding='same', name='block3_conv11')(x)


x = MaxPooling2D((2, 2), strides=(1,1), name='block2_pool')(x)
x = Conv2D(28, (3,3), activation = 'relu',  padding='same')(x)
xp = Flatten()(x) 

#aux_input = Input(shape=(5,), name='aux_input')


x = keras.layers.concatenate([xp, modelu.output, modelg.output, modelr.output, modeli.output , modelz.output])

x = Dense(256, activation='sigmoid', kernel_initializer = 'uniform')(x)
x = Dense(50, activation='relu', kernel_initializer = 'uniform')(x)
#x = Dense(64, activation='relu', kernel_initializer = 'uniform')(x)


x_out = Dense(1, activation = None)(x)

model = Model(inputs=[main_input, modelu.input, modelg.input, modelr.input, modeli.input, modelz.input], outputs=[x_out])
